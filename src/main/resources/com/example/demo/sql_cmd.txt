
create table ETUDIANT
(
    ID            identity,
    NOM           VARCHAR,
    PRENOM        VARCHAR,
    DATENAISSANCE DATE,
    ADRESSE       VARCHAR,
    TEL           VARCHAR(10),
    EMAIL         VARCHAR unique,
    MDP           VARCHAR,
    PROMO         VARCHAR,
    constraint "ETUDIANT_pk"
        primary key (ID)
);

create table INDISPONIBILITE
(
    ID            identity,
    REMARQUE      VARCHAR,
    JOUR          DATE unique,
    EMAIL         VARCHAR unique,
    constraint "INDISPONIBILITE_pk"
        primary key (ID)
);



create table ENSEIGNANT
(
    ID            identity,
    NOM           VARCHAR,
    PRENOM        VARCHAR,
    DATENAISSANCE DATE,
    ADRESSE       VARCHAR,
    TEL           VARCHAR(10),
    EMAIL         VARCHAR unique,
    MDP           VARCHAR,
    GRADE         VARCHAR,
    constraint "ENSEIGNANT_pk"
        primary key (ID)
);


create table ADMINISTRATEUR
(
    ID            identity,
    NOM           VARCHAR,
    PRENOM        VARCHAR,
    DATENAISSANCE DATE,
    ADRESSE       VARCHAR,
    TEL           VARCHAR(10),
    EMAIL         VARCHAR unique,
    MDP           VARCHAR,
    constraint "ADMINISTRATEUR_pk"
        primary key (ID)
);


create table MODULE
(
    CODEM         VARCHAR,
    LIBELLEMOD    VARCHAR,
    VOLUMEHORAIRE INTEGER,
    constraint "MODULE_pk"
        primary key (CODEM)
);


create table SALLE
(
    CODES         VARCHAR,
    DEPART        VARCHAR,
    NBRPLACE      INTEGER,
    constraint "SALLE_pk"
        primary key (CODES)
);


create table SEANCE
(
    ID            identity,
    CODEM         VARCHAR,
    CODES         VARCHAR,
    CODEENS       VARCHAR,
    TYPESEANCE    ENUM('TP','TD','CM'),
    DATE          DATE,
    HEURED        INTEGER,
    HEUREF        INTEGER,
    constraint "SEANCE_pk"
        primary key (ID),
    constraint "SEANCE_MODULE_fk"
        foreign key (CODEM) references MODULE (CODEM),
    constraint "SEANCE_SALLE_fk"
        foreign key (CODES) references SALLE (CODES),
    constraint "SEANCE_ENSEIGNANT_fk"
        foreign key (CODEENS) references ENSEIGNANT (EMAIL)
);
